        const accountsData = typeof response === 'string' ? JSON.parse(response) : response;
        
        // ���� ������� � ������ ������� (� ������ ��������)
        const account = accountsData.accounts?.find(acc => acc.currency.toUpperCase() === coin.toUpperCase());
        
        if (account && account.available_balance) {
            // ���������� ��������� ������ � ����� ������
            res.json({
                success: true,
                balance: account.available_balance.value,
                total: account.hold.value
            });
        } else {
            res.json({
                success: true,
                balance: '0',
                total: '0'
            });
        }
    } catch (error) {
        console.error('Error getting balance:', error);
        res.status(500).json({
            success: false,
            error: error.message
        });
    }
});
